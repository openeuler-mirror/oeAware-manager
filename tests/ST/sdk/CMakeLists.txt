set(test_sdk_src
    test_env_info.cpp
    test_net_intf.cpp
    test_net_hirq.cpp
    test_pmu.cpp
    main.cpp
)

add_compile_options(-g -fPIC -Wall -Wextra -O2)
add_executable(test_sdk 
    ${test_sdk_src}
)

    enable_asan(test_sdk)
endif()

    ${SDK_INC_PATH}
)
